/**
 * php::hex_base64编码器
 * Create at: 2024/09/02
 */

'use strict';

module.exports = (pwd, data) => {
  let num = Math.floor(Math.random() * 15);
  let randomStr = `${Math.random().toString(16).substr(num)}`;

  let hexData = Buffer.from(data['_']).toString('hex');
  let encry = Buffer.from(hexData).toString('base64');

  data['num'] = 15 - num;
  data[pwd] = `${randomStr}` + encry + `${randomStr}`;

  delete data['_'];
  return data;
}
